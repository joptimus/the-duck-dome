Font scale is NOT tokenized into CSS vars (too many size/weight/family combos). Instead, these are the reference values consumed directly:

| Size | Weight | Family | Used for |
|------|--------|--------|----------|
| 22px | 700 | Orbitron | Modal headers, close buttons |
| 20px | 700 | Orbitron | Panel section headers |
| 18px | 700 | Rajdhani | Session type names, large labels |
| 16px | 400-500 | Source Sans 3 | Body text, message content, inputs |
| 15px | 500/700 | Source Sans 3 / Rajdhani | Labels, detail text, agent names |
| 14px | 400-500 | JetBrains Mono | Code tags, channel names, mono text |
| 13px | 400 | Source Sans 3 | Helper text, secondary labels |
| 12px | 700 | Rajdhani | Status tags, section labels, small UI |
| 11px | 400 | JetBrains Mono | Timestamps (often at 50% opacity) |
| 10px | -- | -- | Micro labels (rare) |

```jsx
  const filterId = `bolt-glow-${size}-${color.replace('#', '')}`;

            </filter>
          </defs>

        </>
      )}

      <path d={boltPath} fill="#fff" opacity={0.35} />
    </svg>
  );
}
```

```jsx
import { useMemo } from 'react';
import styles from './Waveform.module.css';

export function Waveform({ color = "#00D4FF", bars = 16 }) {
  // Memoize random durations so they don't change on re-render
  const barConfigs = useMemo(
    () => Array.from({ length: bars }, (_, i) => ({
      duration: 0.3 + Math.random() * 0.5,  // 0.3s - 0.8s
      delay: i * 0.04,                        // staggered by index
    })),
    [bars]
  );

  return (
    <div className={styles.container}>
      {barConfigs.map((cfg, i) => (
        <div
          key={i}
          className={styles.bar}
          style={{
            background: color,
            animation: `barPulse ${cfg.duration}s ease-in-out ${cfg.delay}s infinite alternate`,
          }}
        />
      ))}
    </div>
  );
}
```

**Key detail:** The random durations are memoized so the waveform doesn't re-randomize on every render. Each bar has `barPulse` animation with `alternate` direction, creating an oscillating scale effect.

### Task 4.1 — Create `apps/web/src/components/effects/ParticleField.jsx`

Canvas-based particle system. No CSS Module needed (all rendering is canvas API).

**Props:** none (self-contained)

```jsx
import { useRef, useEffect } from 'react';

const PARTICLE_COLORS = ["#00D4FF", "#A855F7", "#C084FC", "#00FFA3"];
const PARTICLE_COUNT = 45;

export function ParticleField() {
  const canvasRef = useRef(null);
  const particles = useRef([]);
  const animFrame = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    let w, h;

    const resize = () => {
      w = canvas.width = canvas.parentElement.offsetWidth;
      h = canvas.height = canvas.parentElement.offsetHeight;
    };
    resize();

    // Initialize 45 particles
    particles.current = [];
    for (let i = 0; i < PARTICLE_COUNT; i++) {
      particles.current.push({
        x: Math.random() * w,
        y: Math.random() * h,
        vx: (Math.random() - 0.5) * 0.35,  // horizontal velocity: -0.175 to +0.175
        vy: (Math.random() - 0.5) * 0.2,    // vertical velocity: -0.1 to +0.1
        s: Math.random() * 2 + 0.5,          // radius: 0.5px to 2.5px
        c: PARTICLE_COLORS[~~(Math.random() * PARTICLE_COLORS.length)],
        a: Math.random() * 0.3 + 0.05,       // base alpha: 0.05 to 0.35
        p: Math.random() * Math.PI * 2,       // phase: 0 to 2*PI
        ps: 0.008 + Math.random() * 0.012,    // phase speed: 0.008 to 0.020
      });
    }

    const draw = () => {
      ctx.clearRect(0, 0, w, h);

      particles.current.forEach((p) => {
        // Update position
        p.x += p.vx;
        p.y += p.vy;
        p.p += p.ps;

        // Wrap around edges
        if (p.x < 0) p.x = w;
        if (p.x > w) p.x = 0;
        if (p.y < 0) p.y = h;
        if (p.y > h) p.y = 0;

        // Oscillating alpha (sine wave)
        const alpha = p.a * (0.5 + 0.5 * Math.sin(p.p));

        // Draw particle core
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.s, 0, Math.PI * 2);
        ctx.fillStyle = p.c;
        ctx.globalAlpha = alpha;
        ctx.fill();

        // Draw glow halo (3.5x radius, 7% of particle alpha)
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.s * 3.5, 0, Math.PI * 2);
        ctx.globalAlpha = alpha * 0.07;
        ctx.fill();
      });

      ctx.globalAlpha = 1;
      animFrame.current = requestAnimationFrame(draw);
    };

    animFrame.current = requestAnimationFrame(draw);
    window.addEventListener('resize', resize);

    return () => {
      cancelAnimationFrame(animFrame.current);
      window.removeEventListener('resize', resize);
    };
  }, []);

  return (
    <canvas
      ref={canvasRef}
      style={{
        position: 'absolute',
        inset: 0,
        pointerEvents: 'none',
        zIndex: 0,
      }}
    />
  );
}
```

**Key implementation details:**
- Exactly 45 particles, not configurable
- 4 colors: blue (`#00D4FF`), purple (`#A855F7`), purpleGlow (`#C084FC`), success (`#00FFA3`)
- Velocity is slow and drifty: max 0.175px/frame horizontal, 0.1px/frame vertical
- Alpha oscillates via sine wave using the particle's phase, creating a "breathing" effect
- Each particle has a glow halo at 3.5x its radius drawn at 7% of its current alpha
- Canvas resizes to parent element dimensions on window resize
- Animation frame is properly cancelled on unmount
- Canvas has `pointer-events: none` and `z-index: 0` so it sits behind all UI

CSS-driven lightning pulse that travels along panel edges.

**Props:** `{ vertical = true, color = "#00D4FF", minDelay = 3000, maxDelay = 7000 }`

```css
.pulseHorizontal {

}
```

**ElectricPulse.jsx:**
```jsx
import { useState, useEffect } from 'react';
import styles from './ElectricPulse.module.css';

export function ElectricPulse({
  vertical = true,
  color = "#00D4FF",
  minDelay = 3000,
  maxDelay = 7000,
}) {
  const [key, setKey] = useState(0);
  const [duration, setDuration] = useState(1.2);

  useEffect(() => {
    let timeout;
    const schedule = () => {
      const delay = minDelay + Math.random() * (maxDelay - minDelay);
      timeout = setTimeout(() => {
        setDuration(0.8 + Math.random() * 0.8);  // 0.8s to 1.6s
        setKey((k) => k + 1);
        schedule();
      }, delay);
    };
    schedule();
    return () => clearTimeout(timeout);
  }, [minDelay, maxDelay]);

  const containerClass = `${styles.container} ${vertical ? styles.vertical : styles.horizontal}`;
  const pulseClass = `${styles.pulse} ${vertical ? styles.pulseVertical : styles.pulseHorizontal}`;
  const animationName = vertical ? 'pulseTravelDown' : 'pulseTravelRight';

  return (
    <div className={containerClass}>
      <div
        key={key}
        className={pulseClass}
        style={{
          background: `radial-gradient(ellipse at center, ${color} 0%, ${color}60 30%, transparent 70%)`,
          boxShadow: `0 0 12px ${color}60, 0 0 24px ${color}25`,
          animation: key > 0 ? `${animationName} ${duration}s ease-out forwards` : 'none',
          opacity: key > 0 ? 1 : 0,
        }}
      />
    </div>
  );
}
```

**Key implementation details:**
- Two modes: **vertical** (travels top-to-bottom along right edge of panels) and **horizontal** (travels left-to-right along bottom edge of top bar)
- Vertical: container is 3px wide, pulse is 40px tall
- Horizontal: container is 3px tall, pulse is 60px wide
- The pulse fires at random intervals between `minDelay` and `maxDelay` milliseconds
- Each pulse has a random duration between 0.8s and 1.6s
- Uses `key` prop trick to re-trigger CSS animation: incrementing the key forces React to unmount/remount the pulse div, restarting the animation
- The pulse has a radial gradient center with a box-shadow glow
- On first render (`key === 0`), the pulse is invisible (`opacity: 0`, `animation: none`)
- The `pulseTravelDown` and `pulseTravelRight` keyframes are defined in `tokens/animations.css` (PR UI-1)
- Typical usage in the app:
  - Top bar bottom edge: `<ElectricPulse vertical={false} color="#A855F7" minDelay={5000} maxDelay={12000} />`
  - Panel left edges: `<ElectricPulse vertical color="#A855F7" minDelay={4000} maxDelay={10000} />`
